✅ 1. Tailwind install করো

✅ 2. Tailwind config generate করো
npx tailwindcss init -p

->create two file->
tailwind.config.js
postcss.config.js

/** @type {import('tailwindcss').Config} */

  content: [
    "./index.html",
    "./src/**/*.{js,ts,jsx,tsx}",
  ],
  theme: {
    extend: {
      colors: {
        primary: '#E58411',
        "primary-light": "#E58411",
      },
    },
  },
  plugins: [],
}
